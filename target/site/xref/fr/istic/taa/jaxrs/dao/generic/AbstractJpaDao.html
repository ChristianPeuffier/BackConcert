1   package fr.istic.taa.jaxrs.dao.generic;
2   
3   import java.io.Serializable;
4   import java.util.List;
5   
6   import jakarta.persistence.EntityManager;
7   import jakarta.persistence.EntityTransaction;
8   
9   public abstract class AbstractJpaDao<K, T extends Serializable> implements IGenericDao<K, T> {
10  
11      /**
12       *  The class of the entity.
13       */
14      private Class<T> clazz;
15  
16      /**
17       * The entity manager.
18       */
19      private final EntityManager entityManager;
20  
21      /**
22       * Constructor.
23       * @param clazs the class of the entity
24       */
25      protected AbstractJpaDao(final Class<T> clazs) {
26          this.entityManager = EntityManagerHelper.getEntityManager();
27          this.clazz = clazs;
28      }
29  
30      /**
31       * Set the class of the entity to set.
32       * @param clazzToSet the class of the entity to set
33       */
34      public void setClazz(final Class<T> clazzToSet) {
35          this.clazz = clazzToSet;
36      }
37  
38      /**
39       * Find an entity by its id.
40       * @param id the id of the entity
41       * @return the entity of type T
42       */
43      public T findOne(final K id) {
44          return entityManager.find(clazz, id);
45      }
46  
47      /**
48       * Find all entities.
49       * @return a list of entities of type T
50       */
51      public List<T> findAll() {
52          return entityManager.createQuery("select e from " + clazz.getName() + " as e", clazz).getResultList();
53      }
54  
55      /**
56       * Save an entity.
57       * @param entity the entity to save
58       */
59      public void save(final T entity) {
60          EntityTransaction t = this.entityManager.getTransaction();
61          t.begin();
62          entityManager.persist(entity);
63          t.commit();
64      }
65  
66      /**
67       * Update an entity.
68       * @param entity the entity to update
69       * @return the updated entity
70       */
71      public T update(final T entity) {
72          EntityTransaction t = this.entityManager.getTransaction();
73          t.begin();
74          T res = entityManager.merge(entity);
75          t.commit();
76          return res;
77      }
78  
79      /**
80       * Delete an entity.
81       * @param entity the entity to delete
82       */
83      public void delete(final T entity) {
84          EntityTransaction t = this.entityManager.getTransaction();
85          t.begin();
86          entityManager.remove(entity);
87          t.commit();
88      }
89  
90      /**
91       * Delete an entity by its id.
92       * @param entityId the id of the entity to delete
93       */
94      public void deleteById(final K entityId) {
95          T entity = findOne(entityId);
96          delete(entity);
97      }
98  }
